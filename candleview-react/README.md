## Install

```bash
npm install --save candleview-react
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'candleview-react'
import 'candleview-react/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
