function generateSampleData(count, basePrice = 100) {
  const data = [];
  const baseTime = Math.floor(Date.now() / 1000) - count * 86400; // ä» count å¤©å‰å¼€å§‹

  let currentPrice = basePrice;

  for (let i = 0; i < count; i++) {
    const open = currentPrice;
    const high = open * (1 + Math.random() * 0.05);
    const low = open * (1 - Math.random() * 0.04);
    const close = low + Math.random() * (high - low);
    const volume = Math.floor(Math.random() * 1000000) + 500000;

    data.push({
      time: baseTime + i * 86400, // æ—¥æ•°æ®
      open: parseFloat(open.toFixed(2)),
      high: parseFloat(high.toFixed(2)),
      low: parseFloat(low.toFixed(2)),
      close: parseFloat(close.toFixed(2)),
      volume: volume,
    });

    currentPrice = close;
  }

  return data;
}

function CompleteExample() {

  const generateData = () => {
    const data = [];
    let currentTime = Math.floor(Date.now() / 1000) - 50 * 86400;

    for (let i = 0; i < 50; i++) {

      <CandleView
        title="TSLA - Tesla Inc."

        timeframe="1d"
